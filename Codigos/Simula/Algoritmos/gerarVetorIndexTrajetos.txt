\Inicio{
	indexTrajetos = lista\_vazia() \\
	desl = 0 \\
	adiciona(desl, indexTrajetos) \\
  
  \Comentario{Geração dos índices para os trajetos das crianças} \\
	\Para{cada trajeto $t$ em trajetosCrianca} {
	      desl += tamanho($t$) - 1 \\
	      adiciona(desl, indexTrajetos) \\
	}
  
  \Comentario{Geração dos índices para os trajetos dos jovens} \\
	\Para{cada trajeto $t$ em trajetosJovem} {
	      desl += tamanho($t$) - 1 \\
	      adiciona(desl, indexTrajetos) \\
	}
  
  \Comentario{Geração dos índices para os trajetos dos adultos} \\
	\Para{cada trajeto $t$ em trajetosAdulto} {
	      desl += tamanho($t$) - 1 \\
	      adiciona(desl, indexTrajetos) \\
	}
  
  \Comentario{Geração dos índices para os trajetos dos idosos} \\
	\Para{cada trajeto $t$ em trajetosIdoso} {
	      desl += tamanho($t$) - 1 \\
	      adiciona(desl, indexTrajetos) \\
	}
	
	nTrajetos = tamanho(trajetosCrianca) + tamanho(trajetosJovem) + tamanho(trajetosAdulto) + tamanho(trajetosIdoso) \\
	
  \Comentario{Geração do vetor de trajetos} \\
	vetorTrajetos = lista\_vazia() \\
	\Para{$i$ de $0$ ate indexTrajetos[tamanho(indexTrajetos) - 1]} {
	      adiciona($i$, vetorTrajetos) \\
	}
}
