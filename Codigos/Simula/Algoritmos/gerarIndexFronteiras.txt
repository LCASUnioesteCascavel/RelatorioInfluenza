\Inicio{
	indexFronteiras = lista\_vazia() \\
	desl = 0 \\
	\Para{cada quadra $q$} {
	      adiciona(desl, indexFronteiras) \\
	      \Para{cada lote $l$ de $q$} {
              \Comentario{Os índices somente são calculados para os lotes} \\
              \Se{$q$ $!=$ "0000"} {
                    \Comentario{Cada vértice de fronteira ocupará três inteiros, por isso a multiplicação por três} \\
                    desl += tamanho(fronteiras[$q$, $l$]) * 3 \\
              }
              adiciona(desl, indexFronteiras) \\
	      }
	}
}
