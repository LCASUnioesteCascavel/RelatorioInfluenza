\Inicio{
	indexCentrosEsquinas = lista\_vazia() \\
	desl = 0 \\
	adiciona(desl, indexCentrosEsquinas) \\
	\Para{cada rua $r$} {
        \Comentario{Cada centro de esquina ocupará três inteiros, por isso a multiplicação por três} \\
	      desl += tamanho(centrosEsquinas[$r$]) * 3 \\
	      adiciona(desl, indexCentrosEsquinas) \\
	}
}
