\Inicio{
	caminho = lista\_vazia() \\
	adiciona(verticeAtual, caminho) \\
  \Comentario{Constrói o caminho partindo do vértice de destino} \\
	\Enqto{contem(verticesAnteriores, verticeAtual)} {
	      verticeAtual = verticesAnteriores[verticeAtual] \\
	      \Se{verticeAtual $!=$ nulo} {
              adicionaInicio(verticeAtual, caminho) \\
	      }
	}
	\Retorna{caminho}
}
