\Inicio{
	indexEsquinas = lista\_vazia() \\
	desl = 0 \\
	adiciona(desl, indexEsquinas) \\
	\Para{cada esquina $e$ em esquinas} {
        \Comentario{Cada vértice de esquina ocupará três inteiros, por isso a multiplicação por três} \\
	      desl += tamanho($e$) * 3 \\
	      adiciona(desl, indexEsquinas) \\
	}
}
