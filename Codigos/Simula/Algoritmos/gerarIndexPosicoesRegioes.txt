\Inicio{
	indexPosicoesRegioes = {0, 0, 0} \\
	
	\Para{cada quadra $q$} {
				\Comentario{Primeira letra do identificador de quadra define a região} \\
				regiao = getRegiao(primeiraLetra($q$)) \\
				q = indexPosicoesRegioes[regiao] \\

	      \Para{cada lote $l$ de $q$} {
						q += tamanho(pontos[$q$, $l$]) \\
	      }
				\Comentario{Altera o elemento que está na posição <regiao> para <q> no vetor} \\
        modifica(regiao, q, indexPosicoesRegioes) \\
	}

	\Comentario{Adiciona um elemento com valor zero no início do vetor} \\
  adicionaNaFrente(0, indexPosicoesRegioes) \\
	\Para{$i$ de 1 até tamanho(indexPosicoesRegioes)} {
				q = indexPosicoesRegioes[i] + indexPosicoesRegioes[i - 1] \\
				\Comentario{Altera o elemento que está na posição <i> para <q> no vetor} \\
				modifica(i, q, indexPosicoesRegioes) \\
	}
}
